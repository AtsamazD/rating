     DECLARE                        @gr int,                  /*ИД группы*/
                                    @sem int,               /*№ семестра*/ 
                                    @contr int,              /* ИД вида контроля (одного из рейтингов)*/
					@dt1 datetime, @dt2 datetime
                                    SET @dt1 ='02.10.2015'      /* начальная дата учета рапортичек*/
                                    SET @dt2 ='03.12.2015'       /* конечная дата учета рапортичек*/

SET @gr = 1340--1716
SET @sem = 1
SET @contr = 1 -- 1 -первый рейтинг, 2 - второй рейтинг


select st.student_id, 
	st.student_surname, 
	st.student_firstname, 
	st.student_secondname,  
	--PERS_SURNAME ,
	sub.subject_name,
	--b.Б1 as 'Текущий рейтинг 1',
	b.Б2 as 'Текущий рейтинг контроль',
	b.B as 'Рубежный рейтинг контроль', 
	b.Б as 'Отработки',
	b.[2p] as 'Вторая попытка'
from DEKANAT.ReitUspev b inner join DEKANAT.students st on st.student_id=b.student_id
				inner join INFO.SUBJECT sub on b.subject_id = sub.subject_id
where  (st.student_group_id =@gr ) and 
       (b.sem_id =@sem ) and 
       (b.control_id=@contr)

order by 7-- and uspev_confirm <> 1 
       -- and nagruz_subject_type_id = c.subject_type_id
